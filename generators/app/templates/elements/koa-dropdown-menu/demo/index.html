<<%= tagName %> label="Dinosaurs"> <<%= themeName %>-menu class="dropdown-content" selected="1"> <<%= themeName %>-item>allosaurus-item> <<%= themeName %>-item>brontosaurus-item> <<%= themeName %>-item>carcharodontosaurus-item> <<%= themeName %>-item>diplodocus-item> -menu> >
